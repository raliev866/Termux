```python
import os
from docx import Document
from docx.shared import Pt, RGBColor
from docx.enum.text import WD_ALIGN_PARAGRAPH

input_path = "/sdcard/Documents/elegant.docx"

def add_page_number(run):
    """–í—Å—Ç–∞–≤–∫–∞ –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–æ–≥–æ –ø–æ–ª—è –Ω–æ–º–µ—Ä–∞ —Å—Ç—Ä–∞–Ω–∏—Ü—ã (PAGE)"""
    fldChar1 = OxmlElement('w:fldChar')
    fldChar1.set(ns.qn('w:fldCharType'), 'begin')
    instrText = OxmlElement('w:instrText')
    instrText.set(ns.qn('xml:space'), 'preserve')
    instrText.text = "PAGE"
    fldChar2 = OxmlElement('w:fldChar')
    fldChar2.set(ns.qn('w:fldCharType'), 'end')
    run._r.append(fldChar1)
    run._r.append(instrText)
    run._r.append(fldChar2)

def apply_book_theme():
    if not os.path.exists(input_path):
        print(f"‚ùå –§–∞–π–ª {input_path} –Ω–µ –Ω–∞–π–¥–µ–Ω! –£–±–µ–¥–∏—Å—å, —á—Ç–æ –æ–Ω –ª–µ–∂–∏—Ç –≤ –ø–∞–ø–∫–µ Documents.")
        return

    print("üöÄ –ù–∞—á–∏–Ω–∞—é –ø–æ–ª–Ω—É—é –∏–∑–¥–∞—Ç–µ–ª—å—Å–∫—É—é –ø–æ–¥–≥–æ—Ç–æ–≤–∫—É —à–∞–±–ª–æ–Ω–∞...")
    doc = Document(input_path)
    styles = doc.styles

    def set_font(style_obj, font_name, size, bold=False, italic=False, color=(0,0,0)):
        f = style_obj.font
        f.name = font_name
        f.size = Pt(size)
        f.bold = bold
        f.italic = italic
        f.color.rgb = RGBColor(*color)

    # 1. –¢–ò–¢–£–õ–¨–ù–´–ô –õ–ò–°–¢ –ò –ó–ê–ì–û–õ–û–í–ö–ò (Montserrat)
    # Title
    if 'Title' in styles:
        s = styles['Title']
        set_font(s, 'Montserrat', 26, bold=True)
        s.paragraph_format.alignment = WD_ALIGN_PARAGRAPH.CENTER
        s.paragraph_format.space_after = Pt(24)
    
    # Subtitle
    if 'Subtitle' in styles:
        s = styles['Subtitle']
        set_font(s, 'Montserrat', 16, color=(60, 60, 60))
        s.paragraph_format.alignment = WD_ALIGN_PARAGRAPH.CENTER
        s.paragraph_format.line_spacing = 1.15

    # Headings 1-3
    h_params = {
        'Heading 1': (18, True, 24, 12),
        'Heading 2': (16, True, 18, 6),
        'Heading 3': (14, False, 12, 6)
    }
    for name, (size, bold, before, after) in h_params.items():
        if name in styles:
            s = styles[name]
            set_font(s, 'Montserrat', size, bold=bold)
            s.paragraph_format.space_before = Pt(before)
            s.paragraph_format.space_after = Pt(after)

    # 2. –û–°–ù–û–í–ù–û–ô –¢–ï–ö–°–¢ (PT Serif / Charter)
    if 'Normal' in styles:
        s = styles['Normal']
        set_font(s, 'PT Serif', 12)
        fmt = s.paragraph_format
        fmt.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY
        fmt.line_spacing = 1.35  # –ò–¥–µ–∞–ª—å–Ω—ã–π –∫–Ω–∏–∂–Ω—ã–π –∏–Ω—Ç–µ—Ä–≤–∞–ª
        fmt.first_line_indent = Pt(18) # –ö—Ä–∞—Å–Ω–∞—è —Å—Ç—Ä–æ–∫–∞ 0.63 —Å–º

    # 3. –ù–û–ú–ï–†–ê –°–¢–†–ê–ù–ò–¶ (Montserrat, –í–Ω–∏–∑—É –ø–æ —Ü–µ–Ω—Ç—Ä—É)
    section = doc.sections[0]
    section.footer_distance = Pt(20)
    footer = section.footer
    footer_para = footer.paragraphs[0] if footer.paragraphs else footer.add_paragraph()
    footer_para.alignment = WD_ALIGN_PARAGRAPH.CENTER
    run = footer_para.add_run()
    add_page_number(run)
    run.font.name = 'Montserrat'
    run.font.size = Pt(10)
    run.font.color.rgb = RGBColor(100, 100, 100)

    # 4. –¢–ê–ë–õ–ò–¶–´ (–ú–∏–Ω–∏–º–∞–ª–∏–∑–º –±–µ–∑ –≤–µ—Ä—Ç–∏–∫–∞–ª—å–Ω—ã—Ö –ª–∏–Ω–∏–π)
    for table in doc.tables:
        try:
            table.style = 'Table Grid'
        except:
            pass
        for row in table.rows:
            for cell in row.cells:
                for para in cell.paragraphs:
                    para.paragraph_format.first_line_indent = Pt(0) # –£–±–∏—Ä–∞–µ–º –∫—Ä–∞—Å–Ω—É—é —Å—Ç—Ä–æ–∫—É –≤ —Ç–∞–±–ª–∏—Ü–∞—Ö
                    for r in para.runs:
                        r.font.name = 'PT Serif'
                        r.font.size = Pt(10)

    # 5. –ö–û–î –ò –ú–û–ù–û–®–ò–†–ò–ù–ù–´–ô –¢–ï–ö–°–¢ (JetBrains Mono)
    for s_name in ['Source Code', 'Verbatim', 'Body Text Char']:
        if s_name in styles:
            set_font(styles[s_name], 'JetBrains Mono', 10, color=(40, 40, 40))

    # 6. –°–ù–û–°–ö–ò –ò –ü–û–î–ü–ò–°–ò
    if 'Footnote Text' in styles:
        set_font(styles['Footnote Text'], 'PT Serif', 9)
    if 'Caption' in styles:
        s = styles['Caption']
        set_font(s, 'Montserrat', 9, italic=True)
        s.paragraph_format.alignment = WD_ALIGN_PARAGRAPH.CENTER

    doc.save(output_path)
    print(f"‚úÖ –ì–æ—Ç–æ–≤–æ! –ò—Ç–æ–≥–æ–≤—ã–π —à–∞–±–ª–æ–Ω —Å–æ—Ö—Ä–∞–Ω–µ–Ω: {output_path}")

if __name__ == "__main__":
    apply_book_theme()

```
