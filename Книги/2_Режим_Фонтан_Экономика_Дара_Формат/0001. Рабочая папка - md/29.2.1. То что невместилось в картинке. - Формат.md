Бизнес аналитика

Пример аналитического запроса для оценки эффективности ячеек

SELECT

> Cell_id,
>
> COUNT(\*) as total_transactions,
>
> AVG(satisfaction_score) as avg_satisfaction,
>
> SUM(contribution_value) as total_value_created,
>
> COUNT(DISTINCT participant_id) as active_participants

FROM contributions

WHERE timestamp \>= NOW() -- INTERVAL '30 days'

GROUP BY cell_id

ORDER BY total_value_created DESC;

> Mapping(address =\> Reputation) public reputations;
>
> Function updateReputation(
>
> Address \_user,
>
> Uint256 \_contributionValue,
>
> Uint256 \_complexityMultiplier
>
> ) public onlyVerifiedContract {
>
> Reputation storage rep = reputations\[\_user\];
>
> Rep.score += \_contributionValue \* \_complexityMultiplier;
>
> Rep.lastUpdate = block.timestamp;
>
> Emit ReputationUpdated(\_user, rep.score);
>
> }

}
